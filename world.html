    import crisis;

    public class World {
        public static final boolean pandemic = true;
        public static final boolean systemicRacism = true;
        public static final boolean globalWarming = true;

        private long population;

        public World(int year) {
            population = 7800000000 * Math.exp(1.0136 * year - 2020);
            if (pandemic) checkPandemicResponse();
            System.out.println(Arrays.toString(checkGlobalResources(year)));
        }

        public static void checkPandemicResponse(year) {
            int ventilatorCount = 140000;
            int COVIDCases = 479000;
            if (this.pandemic) {
                ventilatorCount *= Math.pow(2, year - 2019);
                if (!systemicRacism) {
                    COVIDcases *= Math.pow(1.051, year - 2019);
                } else {
                    COVIDcases *= Math.pow(1.16, year - 2019);
                }
            }
            if (COVIDCases >= 1000000 && ventilatorCount < COVIDcases * 0.33) {
                System.err.println("Error: System shutdown!")
            }
        }

        public static int[] checkGlobalResources(int year) {
            int fossilFuels = 100 / Math.pow(2, year - 2019);  
            int renewableEnergy = 250 / Math.pow(1.5, year - 2019);  
            int polarIceCaps = 100;
            if (this.globalWarming) {
                polarIceCaps /= 4;
            }
            int pollution = 100 * Math.pow(2, year - 2019);  
            if (population >= 8 * Math.pow(10, 9) && (fossilFuels <= 5 || 
              renewableEnergy <= 20 || polarIceCaps <= 10 || pollution >= 300)) {
                System.err.println("Error: System overload!");
            }
            return new int[] {fossilFuels, renewableEnergy, polarIceCaps, pollution};
        }
    }
